abstract class Demo{
	public char alpha;
	Demo(){
		alpha='D';
	}
}
public class Tester extends Demo{
	final public void setAlpha(char alpha) {
		this.alpha=alpha;
	}
	final public void getAlpha() {
		System.out.println("alpha = "+alpha);
	}
	public static void main(String[] args) {
		Tester obj=new Tester();
		obj.setAlpha('A');
		obj.getAlpha();
	}
}
 

alpha = A
alpha = D

Observe the following code and Predict the Output:

abstract class SquareRoot {
	SquareRoot(){
		System.out.println("Abstract Class ");
	}
	abstract double calculate(double val);
	public int result() {
		return 1;
	}
}
public class Tester extends SquareRoot{	
	double calculate(double val) {
		return Math.sqrt(val);
	}
	public static void main(String[] args) {
		Tester obj= new Tester();
		System.out.println(obj.calculate(64));
	}
}
 

Error as the result() method is not Overriden in abstract class
Abstract Class 8.0
Error as @Override annotation is not used while overriding the calculate() method
8.0
